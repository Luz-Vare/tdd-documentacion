def emisiones
        auxiliar = 0
        for alimento in @alimentos
                auxiliar += alimento.value.gei
        end
        return auxiliar.round(2)
end

def impacto_huella

	valor_medio = @vtemisiones*1000/self.alimentos.largo

	if valor_medio < 800
		return 1
	elsif  valor_medio > 1200
		return 3
	else
		return 2
	end

end
